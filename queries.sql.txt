SELECT COUNT(DISTINCT order_id) AS total_orders
FROM pizza_sales;
SELECT SUM(total_price) AS total_revenue
FROM pizza_sales;
SELECT SUM(quantity) AS total_pizzas_sold
FROM pizza_sales;
SELECT 
  SUM(total_price) / COUNT(DISTINCT order_id) AS avg_order_value
FROM pizza_sales;
SELECT pizza_category, SUM(total_price) AS revenue
FROM pizza_sales
GROUP BY pizza_category;
SELECT pizza_size, SUM(total_price) AS revenue
FROM pizza_sales
GROUP BY pizza_size;
SELECT pizza_name, SUM(quantity) AS sold
FROM pizza_sales
GROUP BY pizza_name
ORDER BY sold DESC
LIMIT 5;
SELECT pizza_name, SUM(quantity) AS sold
FROM pizza_sales
GROUP BY pizza_name
ORDER BY sold ASC
LIMIT 5;
SELECT order_date, COUNT(DISTINCT order_id) AS orders
FROM pizza_sales
GROUP BY order_date;
SELECT 
  strftime('%m', order_date) AS month,
  SUM(total_price) AS revenue
FROM pizza_sales
GROUP BY month;
SELECT 
  SUM(quantity) / COUNT(DISTINCT order_id) AS avg_pizzas_per_order
FROM pizza_sales;
SELECT pizza_category, SUM(quantity) AS sold
FROM pizza_sales
GROUP BY pizza_category
ORDER BY sold DESC;
